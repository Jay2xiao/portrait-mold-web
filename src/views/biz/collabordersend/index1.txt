<script setup lang="ts">
import {computed, h, onMounted, reactive, ref} from 'vue';
import { useRouter } from 'vue-router';
import {
  NButton,
  NCard,
  NDataTable,
  NForm,
  NFormItem,
  NImage,
  NInput,
  NInputNumber,
  NModal,
  NSelect,
  NSpace,
  NTag,
  useMessage
} from 'naive-ui';

import {
  fetchCollabSourceOrders,
  sendCollabOrder
} from '@/service/api/biz/collab-order';
import { fetchCollabPartnerList as fetchPartners } from '@/service/api/biz/collab';


import BizFileViewer from '@/views/biz/components/BizFileViewer.vue';
import BizFileUpload from '@/views/biz/components/BizFileUpload.vue';

defineOptions({
  name: 'BizCollabOrderSend'
});

const router = useRouter();
const message = useMessage();

const loadingPartners = ref(false);
const loadingSourceOrders = ref(false);
const sending = ref(false);

const showSourceModal = ref(false);
const sourceOrders = ref<any[]>([]);
const receiverOptions = ref<any[]>([]);

const selectedSourceOrder = ref<any | null>(null);
const rawPhotoFileIds = ref<Array<string | number>>([]);
const remarkImageFileIds = ref<Array<string | number>>([]);
const hdPhotoFileIds = ref<Array<string | number>>([]);
const aiModelFileIds = ref<Array<string | number>>([]);

const sendForm = reactive({
  sourceOrderId: null as string | number | null,
  receiverTenantId: '',
  serviceType: 'REPAIR_ONLY',
  materialPackageType: 'RAW_PHOTO',
  senderRepairFeeAmount: null as number | null,
  title: '',
  requirementDesc: '',
  remark: ''
});

const sourceQuery = reactive({
  keyword: '',
  pageNum: 1,
  pageSize: 10
});

const serviceTypeOptions = [
  { label: '只修模', value: 'REPAIR_ONLY' },
  { label: '只打印', value: 'PRINT_ONLY' },
  { label: '修模 + 打印', value: 'REPAIR_PRINT' }
];

const materialPackageTypeOptions = [
  { label: '只有原始照片/备注图', value: 'RAW_PHOTO' },
  { label: '高清照片/备注图', value: 'HD_PHOTO' },
  { label: '高清照片 + AI模型文件/备注图', value: 'HD_PHOTO_AI_MODEL' }
];

function unwrapData(res: any) {
  return res?.data || res;
}

function serviceTypeLabel(value?: string) {
  return serviceTypeOptions.find(item => item.value === value)?.label || value || '-';
}

function materialTypeLabel(value?: string) {
  return materialPackageTypeOptions.find(item => item.value === value)?.label || value || '-';
}

function partnerLabel(row: any) {
  return row.partnerTenantName || row.partnerTenantId || '-';
}

const receiverSelectOptions = computed(() => {
  return receiverOptions.value.map((item: any) => ({
    label: item.partnerTenantName || item.partnerTenantId,
    value: item.partnerTenantId
  }));
});

const sourceColumns = [
  {
    title: '订单号',
    key: 'orderNo',
    width: 180
  },
  {
    title: '客户名称',
    key: 'customerNameSnapshot',
    width: 180
  },
  {
    title: '业务状态',
    key: 'businessStatus',
    width: 120
  },
  {
    title: '应收',
    key: 'receivableAmount',
    width: 100
  },
  {
    title: '未收',
    key: 'unpaidAmount',
    width: 100
  },
  {
    title: '创建时间',
    key: 'createTime',
    width: 170
  },
  {
    title: '操作',
    key: 'actions',
    width: 100,
    render(row: any) {
      return h(
        NButton,
        {
          size: 'small',
          type: 'primary',
          onClick: () => chooseSourceOrder(row)
        },
        { default: () => '选择' }
      );
    }
  }
];

function openSourceModal() {
  showSourceModal.value = true;
  loadSourceOrders();
}

async function loadPartners() {
  loadingPartners.value = true;
  try {
    const res = await fetchPartners({ status: 'ACCEPTED' });
    const list = unwrapData(res) || [];
    receiverOptions.value = Array.isArray(list) ? list : [];
  } finally {
    loadingPartners.value = false;
  }
}

async function loadSourceOrders() {
  loadingSourceOrders.value = true;
  try {
    const res = await fetchCollabSourceOrders({
      keyword: sourceQuery.keyword,
      pageNum: sourceQuery.pageNum,
      pageSize: sourceQuery.pageSize
    });

    const data = unwrapData(res) || {};
    sourceOrders.value = data.rows || [];
  } finally {
    loadingSourceOrders.value = false;
  }
}

function chooseSourceOrder(row: any) {
  selectedSourceOrder.value = row;
  sendForm.sourceOrderId = row.orderId;
  sendForm.title = `${row.orderNo || ''} - ${serviceTypeLabel(sendForm.serviceType)}`;
  showSourceModal.value = false;
}

function buildFiles() {
  const files: any[] = [];

  rawPhotoFileIds.value.forEach(fileId => {
    files.push({
      fileType: 'RAW_PHOTO',
      fileId,
      visibleToSender: '1',
      visibleToReceiver: '1'
    });
  });

  remarkImageFileIds.value.forEach(fileId => {
    files.push({
      fileType: 'REMARK_IMAGE',
      fileId,
      visibleToSender: '1',
      visibleToReceiver: '1'
    });
  });

  hdPhotoFileIds.value.forEach(fileId => {
    files.push({
      fileType: 'HD_PHOTO',
      fileId,
      visibleToSender: '1',
      visibleToReceiver: '1'
    });
  });

  aiModelFileIds.value.forEach(fileId => {
    files.push({
      fileType: 'AI_MODEL_FILE',
      fileId,
      visibleToSender: '1',
      visibleToReceiver: '1'
    });
  });

  return files;
}

function validateBeforeSend() {
  if (!sendForm.sourceOrderId) {
    message.warning('请选择发单订单');
    return false;
  }

  if (!sendForm.receiverTenantId) {
    message.warning('请选择接单方');
    return false;
  }

  if (!sendForm.serviceType) {
    message.warning('请选择服务类型');
    return false;
  }

  if (!sendForm.materialPackageType) {
    message.warning('请选择资料类型');
    return false;
  }

  if (sendForm.serviceType === 'PRINT_ONLY' && sendForm.materialPackageType !== 'HD_PHOTO_AI_MODEL') {
    message.warning('只打印订单必须选择“高清照片 + AI模型文件”资料类型');
    return false;
  }

  if (sendForm.materialPackageType === 'RAW_PHOTO' && rawPhotoFileIds.value.length === 0) {
    message.warning('请上传原始照片');
    return false;
  }

  if (sendForm.materialPackageType === 'HD_PHOTO' && hdPhotoFileIds.value.length === 0) {
    message.warning('请上传高清照片');
    return false;
  }

  if (sendForm.materialPackageType === 'HD_PHOTO_AI_MODEL') {
    if (hdPhotoFileIds.value.length === 0) {
      message.warning('请上传高清照片');
      return false;
    }

    if (aiModelFileIds.value.length === 0) {
      message.warning('请上传AI模型文件');
      return false;
    }
  }

  return true;
}

async function submitSend() {
  if (!validateBeforeSend()) return;

  sending.value = true;

  try {
    await sendCollabOrder({
      sourceOrderId: sendForm.sourceOrderId,
      receiverTenantId: sendForm.receiverTenantId,
      serviceType: sendForm.serviceType,
      materialPackageType: sendForm.materialPackageType,
      senderRepairFeeAmount: sendForm.senderRepairFeeAmount,
      title: sendForm.title,
      requirementDesc: sendForm.requirementDesc,
      remark: sendForm.remark,
      files: buildFiles()
    });

    message.success('协作单已发送');
    router.push('/biz/collaborder');
  } finally {
    sending.value = false;
  }
}


onMounted(() => {
  loadPartners();
});
</script>

<template>
  <NSpace vertical :size="16">
    <NCard title="好友定向发单" :bordered="false">
      <NSpace justify="space-between" align="center">
        <div>
          <strong>发单给好友商家</strong>
          <div style="margin-top: 6px; color: #888">
            先选源订单，再选好友商家，按资料类型上传后发送。
          </div>
        </div>

        <NButton @click="openSourceModal">选择源订单</NButton>
      </NSpace>
    </NCard>

    <NCard title="发单信息" :bordered="false">
      <NForm label-placement="left" label-width="130">
        <NFormItem label="源订单" required>
          <NInput
            :value="selectedSourceOrder?.orderNo || '请选择源订单'"
            readonly
            placeholder="请选择源订单"
          />
        </NFormItem>

        <NFormItem label="接单方好友" required>
          <NSelect
            v-model:value="sendForm.receiverTenantId"
            :options="receiverSelectOptions"
            placeholder="请选择好友商家"
          />
        </NFormItem>

        <NFormItem label="服务类型" required>
          <NSelect
            v-model:value="sendForm.serviceType"
            :options="serviceTypeOptions"
          />
        </NFormItem>

        <NFormItem label="资料类型" required>
          <NSelect
            v-model:value="sendForm.materialPackageType"
            :options="materialPackageTypeOptions"
          />
        </NFormItem>

        <NFormItem label="发单方修模费用（可为空）">
          <NInputNumber
            v-model:value="sendForm.senderRepairFeeAmount"
            :min="0"
            style="width: 220px"
            placeholder="可为空"
          />
        </NFormItem>

        <NFormItem label="标题">
          <NInput v-model:value="sendForm.title" placeholder="可自动生成，也可手动修改" />
        </NFormItem>

        <NFormItem label="备注/要求">
          <NInput
            v-model:value="sendForm.requirementDesc"
            type="textarea"
            :autosize="{ minRows: 3, maxRows: 6 }"
            placeholder="请输入协作要求"
          />
        </NFormItem>

        <NFormItem label="补充备注">
          <NInput
            v-model:value="sendForm.remark"
            type="textarea"
            :autosize="{ minRows: 2, maxRows: 4 }"
            placeholder="备注"
          />
        </NFormItem>
      </NForm>
    </NCard>

    <NCard title="上传资料" :bordered="false">
      <NSpace vertical :size="16">
        <NCard
          v-if="sendForm.materialPackageType === 'RAW_PHOTO'"
          title="原始照片"
          size="small"
        >
          <NSpace vertical>
            <div style="color: #888">
              上传未修复照片，接单方会先处理高清图。
            </div>

            <BizFileUpload
              v-model="rawPhotoFileIds"
              biz-type="TEMP"
              file-stage="COLLAB_SEND"
              file-type="RAW_PHOTO"
              :max="20"
            />

            <BizFileViewer
              v-if="rawPhotoFileIds.length > 0"
              :file-ids="rawPhotoFileIds"
              mode="image"
              :max="20"
              :thumb-size="80"
              show-name
            />
          </NSpace>
        </NCard>

        <NCard
          v-if="sendForm.materialPackageType === 'HD_PHOTO' || sendForm.materialPackageType === 'HD_PHOTO_AI_MODEL'"
          title="高清照片"
          size="small"
        >
          <NSpace vertical>
            <div style="color: #888">
              上传已经修复好的高清照片。
            </div>

            <BizFileUpload
              v-model="hdPhotoFileIds"
              biz-type="TEMP"
              file-stage="COLLAB_SEND"
              file-type="HD_PHOTO"
              :max="20"
            />

            <BizFileViewer
              v-if="hdPhotoFileIds.length > 0"
              :file-ids="hdPhotoFileIds"
              mode="image"
              :max="20"
              :thumb-size="80"
              show-name
            />
          </NSpace>
        </NCard>

        <NCard
          v-if="sendForm.materialPackageType === 'HD_PHOTO_AI_MODEL'"
          title="AI模型文件"
          size="small"
        >
          <NSpace vertical>
            <div style="color: #888">
              上传AI建好的模型文件，如 STL / OBJ / 3MF / ZIP 等。
            </div>

            <BizFileUpload
              v-model="aiModelFileIds"
              biz-type="TEMP"
              file-stage="COLLAB_SEND"
              file-type="AI_MODEL_FILE"
              :max="10"
            />

            <BizFileViewer
              v-if="aiModelFileIds.length > 0"
              :file-ids="aiModelFileIds"
              mode="file"
              :max="10"
              :thumb-size="80"
              show-name
            />
          </NSpace>
        </NCard>

        <NCard title="备注图" size="small">
          <NSpace vertical>
            <div style="color: #888">
              可选，上传备注图、标注图、沟通说明图片。
            </div>

            <BizFileUpload
              v-model="remarkImageFileIds"
              biz-type="TEMP"
              file-stage="COLLAB_SEND"
              file-type="REMARK_IMAGE"
              :max="10"
            />

            <BizFileViewer
              v-if="remarkImageFileIds.length > 0"
              :file-ids="remarkImageFileIds"
              mode="image"
              :max="10"
              :thumb-size="80"
              show-name
            />
          </NSpace>
        </NCard>
      </NSpace>
    </NCard>


    <NSpace justify="end">
      <NButton @click="router.back()">取消</NButton>
      <NButton type="primary" :loading="sending" @click="submitSend">发送协作单</NButton>
    </NSpace>

    <NModal
      v-model:show="showSourceModal"
      preset="card"
      title="选择源订单"
      style="width: 1250px"
    >
      <NForm inline label-placement="left">
        <NFormItem label="关键词">
          <NInput
            v-model:value="sourceQuery.keyword"
            clearable
            placeholder="订单号 / 客户名称"
            style="width: 280px;"
          />
        </NFormItem>

        <NFormItem>
          <NButton type="primary" @click="loadSourceOrders">查询</NButton>
        </NFormItem>
      </NForm>

      <NDataTable
        size="small"
        :loading="loadingSourceOrders"
        :columns="sourceColumns"
        :data="sourceOrders"
        :pagination="false"
        :scroll-x="1200"
      />
    </NModal>
  </NSpace>
</template>
